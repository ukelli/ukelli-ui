## With Icon

<Playground>
  {
    () => {
      const icons = [
        {icon: 'envelope-open', s: 'r'},
        {icon: 'paper-plane', s: 'r'},
        {icon: 'align-right'},
        {icon: 'bell', s: 'r'},
        {icon: 'check-circle', s: 'r'},
      ];
      return (
        <div>
          {
            ForEachColor(ExampleColors)((color, idx) => {
              return (
                <Button key={color}
                  color={color}
                  hola
                  icon={icons[idx].icon}
                  s={icons[idx].s}
                  className="mr10 mb10">
                  {color.toUpperCase()}
                </Button>
              )
            })
          }
        </div>
      )
    }
  }
</Playground>
