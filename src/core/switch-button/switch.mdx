import { useState } from 'react'
import { Playground, Props } from 'docz'
import { Switch } from './';

# Switch

## Basic Usage

<Playground>
  {
    () => {
      const [checked1, setChecked1] = useState(false)
      const [checked2, setChecked2] = useState(0)
      return (
        <div>
          <h4>开关, 输出结果: {JSON.stringify(checked1)}</h4>
          <Switch
            onChange={val => setChecked1(val)}
            checked={checked1} />
        </div>
      )
    }
  }
</Playground>

## 指定期望输出的值

<Playground>
  {
    () => {
      const [checked1, setChecked1] = useState(false)
      const [checked2, setChecked2] = useState(0)
      return (
        <div>
          <h4>输出值: {checked2}</h4>
          <Switch
            onChange={val => {
              console.log(val)
              setChecked2(val)
            }}
            outputs={[1, 0]}
            checked={checked2} />
        </div>
      )
    }
  }
</Playground>
