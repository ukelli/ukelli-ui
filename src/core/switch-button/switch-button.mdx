import { useState } from 'react';
import { Playground, Props } from 'docz'
import { SwitchBtn } from './';

# SwitchBtn

切换按钮

## Basic Usage

<Playground>
  {
    () => {
      const [activeIdx, setIdx] = useState('btn1')

      return (
        <div>
          <SwitchBtn
            onSwitch={val => setIdx(val)}
            activeIdx={activeIdx}
            btns={{
              btn1: '选项1',
              btn2: '选项2',
              btn3: '选项3',
            }} />
        </div>
      )
    }
  }
</Playground>

## 输出 Number 类型的值

<Playground>
  {
    () => {
      const [activeIdx, setIdx] = useState(0)

      return (
        <div>
          <SwitchBtn
            onSwitch={val => setIdx(val)}
            activeIdx={activeIdx}
            isNum
            btns={{
              0: '选项1',
              1: '选项2',
              2: '选项3',
            }} />
        </div>
      )
    }
  }
</Playground>
